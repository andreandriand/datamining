##Menentukan Dissimilarity

def diss(i,j):
  p = 2
  m = 0
  for col in ['category', 'started']:
    if data[col][i] == data[col][j]:
      m+=1
  return (p-m)/p

print(diss(1,2))
print(diss(1,3))
print(diss(7,8))
print(diss(18,21))

##Numeric Attribute : Minkowski Distance

def convert(s):
  return int("".join(s.split(",")))

def diss2(i,j,h):
  result = 0
  for col in ["subscribers", "video views", "video count"]:
    result = math.pow(abs(convert(data[col][i]) - convert(data[col][j])), h)
  return result**(1/h)
